API Documentation

  


  
  

    
GET /sim

    
Retrieve chatbot-style responses to queries.


    

      
      
    


    
    

      

        

const axios = require('axios');
axios.get('https://simsimi.ooguy.com/sim', {
  params: { query: 'hi' }
}).then(res => console.log(res.data));


      

    


    

      

        

import requests
res = requests.get("https://simsimi.ooguy.com/sim", params={"query": "hi"})
print(res.json())


      

    


    

      

        

fetch("https://simsimi.ooguy.com/sim?query=hi")
  .then(res => res.json())
  .then(data => console.log(data));


      

    

  


  
  

    
GET /teach

    
Teach the bot how to respond using a phrase and an answer.


    

      
      
    


    

      

        

const axios = require('axios');
axios.get('https://simsimi.ooguy.com/teach', {
  params: { ask: 'hello', ans: 'hey!' }
}).then(res => console.log(res.data));


      

    


    

      

        

import requests
res = requests.get("https://simsimi.ooguy.com/teach", params={"ask": "hello", "ans": "hey!"})
print(res.json())


      

    


    

      

        

fetch("https://simsimi.ooguy.com/teach?ask=hello&ans=hey!")
  .then(res => res.json())
  .then(data => console.log(data));
